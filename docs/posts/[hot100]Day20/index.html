1 正则表达式匹配

分析

采用动态规划的方法进行求解，关键在于如何确定状态转移方程和状态的定义，我们发现：

  • 当遍历到的正则化表达式的字符不为’*’号时，只需要判断目前的s[i]和p[j]是否相等，若相等则查看s[1..i-1]和p[1..j-1]是否匹配
  • 当遍历到’‘时，有两种情况，’’的前一个字符和目前s[i]不一致，则考虑s[1..i]和p[1..j-2]的匹配情况；当一致时，可以不匹配考虑s[1..i]和p[1..j-2]也可以匹配一个s[1..i-1]和p[1..j]

综上状态转移方程如下：

f[i][j]的含义是，当其为True时，表明s[1..i]和p[1..j]是匹配的；否则不匹配。

  • C++
  • Python
class Solution {
public:
    bool isMatch(string s, string p) {
        class InnerClass {
        public:
            string s;
            string p;
            InnerClass(string &s, string &p) {
                this->s = s;
                this->p = p;
            }
            bool match(int i, int j) {
                if (i == 0) return false;
                return this->p[j-1] == '.' ? true : this->s[i-1] == this->p[j - 1];
            }
        };  // 内部类的声明-为了实现内部函数
        InnerClass Inner(s, p);
        int n = s.size(), m = p.size();
        std::vector<std::vector<bool>> dp(n + 1, std::vector<bool>(m + 1, false));  // 创建bool类型的二维表
        dp[0][0] = true;  // 边界条件
        for (int i = 0; i < n + 1; i++) {
            for (int j = 1; j < m + 1; j++) {
                if (p[j-1] != '*') {
                    dp[i][j] = Inner.match(i, j) ? dp[i-1][j-1] : false;
                }
                else {
                    dp[i][j] = Inner.match(i, j-1) ? (dp[i-1][j] || dp[i][j-2]) : dp[i][j-2];
                }
            }
        }
        return dp[n][m];

    }
};
class Solution:
    def isMatch(self, s: str, p: str) -> bool:
        n = len(s); m = len(p)
        # 创建状态转移矩阵
        dp = [[False]*(m+1) for _ in range(n+1)]  # 当采用0-based索引时无法表示空字符串
        # 边界条件，当两个空字符串进行匹配时结果一定为真
        dp[0][0] = True

        def match(i: int, j: int) -> bool:
            if i == 0:
                return False
            # 只要进行比较就不匹配，应当直接回到dp[0][0]
            return s[i-1] == p[j-1] if p[j-1] != '.' else True
            # i-1,j-1因为索引是1-based

        for i in range(n+1):
            # 当i=0,j!=0时不一定都dismatch s = "", p = "c*"，故i得从0开始遍历
            for j in range(1, m+1):
                # 状态转移方程
                if p[j-1] != '*':
                    dp[i][j] = dp[i-1][j-1] if match(i, j) else False
                else:
                    dp[i][j] = dp[i-1][j] or dp[i][j-2] if match(i, j-1) else dp[i][j-2]
                    # 之所以加入or是因为s = "a", p = "ab*a*"，如果此处将a匹配掉了，则最开始的a将没办法匹配了
        
        return dp[n][m]

2 通配符匹配

分析

该题和上题类似，“?”和上题中的 ‘.’ 等价，而 “*” 的含义有所改变，其不再需要查看其前面的字符，而是直接任意匹配，故有如下转移状态方程：

  • C++
class Solution:
    def isMatch(self, s: str, p: str) -> bool:
        n = len(s); m = len(p)
        dp = [[False] * (m+1) for _ in range(n+1)]
        dp[0][0] = True
        def match(i: int, j: int) -> bool:
            if i == 0: return False
            return True if p[j-1] == "?" else s[i-1] == p[j-1]
        for i in range(n+1):
            for j in range(1, m+1):
                if p[j-1] != '*':
                    dp[i][j] = dp[i-1][j-1] if match(i, j) else False
                else:
                    dp[i][j] = (dp[i-1][j] or dp[i][j-1]) if i != 0 else dp[i][j-1]
        return dp[n][m]  
class Solution {
public:
    bool isMatch(string s, string p) {
        int n = s.size(), m = p.size();
        std::vector<std::vector<bool>> dp(n+1, std::vector<bool>(m+1, false));
        dp[0][0] = true;  // 初始化不要忘了！！
        class InnerClass {
        public:
            string s;
            string p;
            InnerClass(string &s, string &p) {
                this->s = s;
                this->p = p;
            }
            bool match(int i, int j) {
                if (i == 0) return false;
                return this->p[j-1] == '?' ? true : this->s[i-1] == this->p[j-1];
            }
        };
        InnerClass Inner(s, p);
        for (int i = 0; i < n + 1; i++) {
            for (int j = 1; j < m + 1; j++) {
                if (p[j-1] != '*') {
                    dp[i][j] = Inner.match(i,j) ? dp[i-1][j-1] : false; 
                } else {
                    dp[i][j] = (i == 0) ? dp[i][j-1] : (dp[i-1][j] || dp[i][j-1]);
                }
            }
        }
        return dp[n][m];
    }
};
