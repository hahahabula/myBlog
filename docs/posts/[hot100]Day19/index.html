2 最长回文子串

分析

采用动态规划的方法，此处一个回文子串去掉两端相同的字符串也得是回文串，否则这个子串就不是回文串。

  • C++
  • Python
class Solution {
public:
    string longestPalindrome(string s) {
        int n = s.size();
        if (n < 2) return s;
        int begin = 0, max_len = 1;
1        bool dp[n][n];

        for (int j = 0; j < n; j++) {
            dp[j][j] = true;
            for (int i = 0; i < j; i++) {
                dp[i][j] = s[i] == s[j] && (j - i < 2 || dp[i+1][j-1]);
                if (dp[i][j] && j - i + 1 > max_len) {
                    begin = i;
                    max_len = j - i + 1;
                }
            }
        }
        return s.substr(begin, max_len);
    }
};

class Solution:
    def longestPalindrome(self, s: str) -> str:
        n = len(s)
        # 边界条件当字符串不足两个时必为回文
        if n < 2:
            return s
        # 记录最大长度和起始点
        max_len = 1
        begin = 0
        # d[i][j]表示s[i..j]是否为回文串
        dp = [[False] * n for _ in range(n)]
        for j in range(n):  # 得按列对状态转移矩阵赋值-确保左下角的元素已经取值
            # 对角线上一定是回文串
            dp[j][j] = True
            for i in range(0, j):
                # 状态转移方程
                dp[i][j] = (s[i] == s[j] and (j - i < 2 or dp[i+1][j-1]))
                # 边界条件 j - i < 2
                if dp[i][j] and (j - i + 1 > max_len):
                    max_len = j - i + 1
                    begin = i
        return s[begin:begin+max_len]
